    def compute_d3D(self):
        #  Computes 3D distances between every possible node pair . dimension:(numSnapShots,numNodes,numNodes)
        nodeloc   = np.concatenate((self.locBS, self.locUE), axis= 1, dtype = np.float32)   # node locations across snapshots 
        numOfSnapShots, numNodes  = nodeloc.shape[0:2]

        return np.linalg.norm(nodeloc.repeat(numNodes,axis=-2) - np.tile(nodeloc,(1,numNodes,1)), 
                              axis=-1).reshape(numOfSnapShots, numNodes, numNodes)

        return np.linalg.norm(nodeloc.repeat(numNodes,axis=-2) - np.tile(nodeloc,(1,numNodes,1)), 
                              axis=-1).reshape(numOfSnapShots, numNodes, numNodes)

    def generateSCMParameters(self, muLog10 = -7.5, sigmaLog10=0.4):
        numClustersMax = self.numClusters.max()
        numClustersMin = self.numClusters.min()
        pathDelay = np.zeros((self.fc.size, self.numOfSnapShots, self.numOfBS, 
                              self.numOfUE, numClustersMax+4), dtype=np.float64)

        nlosBias = 10**(muLog10 + sigmaLog10*np.random.normal(size = pathDelay[0][~self.linkStateVec_BS_UE].shape))

        for nc in range(self.fc.size):

            # Delay Insertion

            ## LoS Delay Insertion

            delays = self.pathDelay[nc][self.linkStateVec_BS_UE]

            i      = np.argwhere(self.linkStateVec_BS_UE[..., np.newaxis])
            i[:,3] = 0
            index  = tuple(np.array(list(zip(*i))))
            pathDelay[nc][index]= delays[...,0] + self.d3D[:, 0:self.numOfBS, self.numOfBS::][self.linkStateVec_BS_UE]/(3*10**8)

            i = np.argwhere(self.linkStateVec_BS_UE[..., np.newaxis].repeat(3,-1))
            i[:,3] = i[:,3] + 1
            index  = tuple(np.array(list(zip(*i))))
            pathDelay[nc][index] = (delays[...,1, np.newaxis] + (self.cDS[nc][self.linkStateVec_BS_UE].reshape(-1,1))*np.array([0,1.28, 2.56])*10**-9 + self.d3D[:, 0:self.numOfBS, self.numOfBS::, np.newaxis][self.linkStateVec_BS_UE]/(3*10**8)).flatten()

            i[:,3] = i[:,3] + 3
            index  = tuple(np.array(list(zip(*i))))
            pathDelay[nc][index] = (delays[...,2, np.newaxis] + (self.cDS[nc][self.linkStateVec_BS_UE].reshape(-1,1))*np.array([0,1.28, 2.56])*10**-9 + self.d3D[:, 0:self.numOfBS, self.numOfBS::, np.newaxis][self.linkStateVec_BS_UE]/(3*10**8)).flatten()

            i = np.argwhere(self.linkStateVec_BS_UE[..., np.newaxis].repeat(numClustersMin-3,-1))
            i[:,3] = i[:,3] + 7
            index  = tuple(np.array(list(zip(*i))))
            pathDelay[nc][index] = (delays[...,3:numClustersMin] + self.d3D[:, 0:self.numOfBS, self.numOfBS::][self.linkStateVec_BS_UE, np.newaxis]/(3*10**8)).flatten()

            ## NLoS Delay Insertion

            delays = self.pathDelay[nc][~self.linkStateVec_BS_UE]

    def getTDLchannel(self):
        lamda = 3*10**8/self.fc 
        hchannel = np.zeros((lamda.size, self.numOfSnapShots, self.numOfBS, self.numOfUE, 
                             self.numClusters, self.ueAntennaArray[0].numAntennas, 
                             self.bsAntennaArray[0].numAntennas,),
                            dtype=np.complex64)
        numCarriers = np.minimum(self.bsAntennaArray.size, self.ueAntennaArray.size)
        
        if(numCarriers != lamda.size):
            if(self.bsAntennaArray.size == 1):
                self.bsAntennaArray = self.bsAntennaArray.repeat(lamda.size)
            else:
                self.ueAntennaArray = self.ueAntennaArray.repeat(lamda.size)
        
        for nc in range(lamda.size):
            txAntLocation = self.locBS[...,np.newaxis,:] + self.bsAntennaArray[nc].locations.reshape(-1,3)
            rxAntLocation = self.locUE[...,np.newaxis,:] + self.ueAntennaArray[nc].locations.reshape(-1,3)

            ## Compute Reciever Stearing Vector
            rRx = self.drc(self.thetaAoA_mn, self.phiAoA_mn)[..., np.newaxis, np.newaxis, :]
            dRx = rxAntLocation[:, np.newaxis, :, np.newaxis, np.newaxis, ..., np.newaxis]

            ## Compute Transmitter Stearing Vector 
            rTx = self.drc(self.thetaAoD_mn, self.phiAoD_mn)[..., np.newaxis, np.newaxis, :]
            dTx = txAntLocation[:, :, np.newaxis, np.newaxis, np.newaxis, ..., np.newaxis]

            ## Apply Doppler Phasors: Rx Stearing Vector + Tx Stearing Vector
            hn  = np.complex64(np.sqrt(self.pathPower/self.numRaysPerCluster)*np.exp(1j*2*np.pi*((rRx@dRx)[..., 0]+(rTx@dTx)[..., np.newaxis, :, 0, 0])/lamda[nc]))

            ## Apply Doppler Phasors: Rx Mobility + Tx Mobility

    def drc(self, theta, phi):
        x = np.sin(theta*np.pi/180)*np.cos(phi*np.pi/180)
        y = np.sin(theta*np.pi/180)*np.sin(phi*np.pi/180)
        z = np.cos(theta*np.pi/180)
        return np.stack((x,y,z), axis=-1)

    def AoD(self, txLocations, rxLocations):
        numTx = txLocations.shape[-2]
        numRx = rxLocations.shape[-2]
        numSnapshots = rxLocations.shape[0]
        drcs  = np.tile(rxLocations, (1, numTx, 1)) - txLocations.repeat(numRx, axis = -2)
        e     = np.arctan2(np.linalg.norm(drcs[...,0:2], axis=-1), drcs[...,2]).reshape(numSnapshots, numTx, numRx)
        a     = np.arctan2(drcs[...,1], drcs[...,0]).reshape(numSnapshots, numTx, numRx)
        return np.stack([e, a], axis = -1)
